import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class QuizApp {
    // Question class to store question details
    static class Question {
        private String questionText;
        private String[] options;
        private int correctAnswer;
        
        public Question(String questionText, String[] options, int correctAnswer) {
            this.questionText = questionText;
            this.options = options;
            this.correctAnswer = correctAnswer;
        }
        
        public String getQuestionText() {
            return questionText;
        }
        
        public String[] getOptions() {
            return options;
        }
        
        public int getCorrectAnswer() {
            return correctAnswer;
        }
    }
    
    private List<Question> questions;
    private int score;
    
    public QuizApp() {
        questions = new ArrayList<>();
        score = 0;
        initializeQuestions();
    }
    
    // Initialize sample questions
    private void initializeQuestions() {
        questions.add(new Question(
            "What is the capital of France?",
            new String[]{"1. London", "2. Paris", "3. Berlin", "4. Madrid"},
            2
        ));
        questions.add(new Question(
            "Which planet is known as the Red Planet?",
            new String[]{"1. Venus", "2. Mars", "3. Jupiter", "4. Saturn"},
            2
        ));
        questions.add(new Question(
            "What is 2 + 2?",
            new String[]{"1. 3", "2. 4", "3. 5", "4. 6"},
            2
        ));
        questions.add(new Question(
            "Which programming language is known for its use in web development?",
            new String[]{"1. Python", "2. Java", "3. JavaScript", "4. C++"},
            3
        ));
        questions.add(new Question(
            "What is the largest ocean on Earth?",
            new String[]{"1. Atlantic Ocean", "2. Indian Ocean", "3. Arctic Ocean", "4. Pacific Ocean"},
            4
        ));
    }
    
    // Start the quiz
    public void startQuiz() {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Welcome to the Quiz Application!");
        System.out.println("Answer the following questions by entering the option number (1-4).");
        
        for (int i = 0; i < questions.size(); i++) {
            Question q = questions.get(i);
            System.out.println("\nQuestion " + (i + 1) + ": " + q.getQuestionText());
            for (String option : q.getOptions()) {
                System.out.println(option);
            }
            
            System.out.print("Your answer (1-4): ");
            int userAnswer = 0;
            try {
                userAnswer = scanner.nextInt();
                if (userAnswer < 1 || userAnswer > 4) {
                    System.out.println("Invalid input! Please enter a number between 1 and 4.");
                    i--; // Repeat the question
                    continue;
                }
            } catch (Exception e) {
                System.out.println("Invalid input! Please enter a number between 1 and 4.");
                scanner.nextLine(); // Clear invalid input
                i--; // Repeat the question
                continue;
            }
            
            if (userAnswer == q.getCorrectAnswer()) {
                System.out.println("Correct!");
                score++;
            } else {
                System.out.println("Incorrect! The correct answer was option " + q.getCorrectAnswer());
            }
        }
        
        // Display final score
        System.out.println("\nQuiz Completed!");
        System.out.println("Your final score: " + score + "/" + questions.size());
        double percentage = (double) score / questions.size() * 100;
        System.out.printf("Percentage: %.2f%%\n", percentage);
        
        // Provide feedback based on score
        if (percentage >= 80) {
            System.out.println("Excellent performance!");
        } else if (percentage >= 60) {
            System.out.println("Good job!");
        } else {
            System.out.println("Keep practicing!");
        }
        
        scanner.close();
    }
    
    public static void main(String[] args) {
        QuizApp quiz = new QuizApp();
        quiz.startQuiz();
    }
}
